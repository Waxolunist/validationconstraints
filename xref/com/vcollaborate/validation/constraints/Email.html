20  package com.vcollaborate.validation.constraints;
21  
22  import static java.lang.annotation.ElementType.ANNOTATION_TYPE;
23  import static java.lang.annotation.ElementType.CONSTRUCTOR;
24  import static java.lang.annotation.ElementType.FIELD;
25  import static java.lang.annotation.ElementType.METHOD;
26  import static java.lang.annotation.ElementType.PARAMETER;
27  import static java.lang.annotation.RetentionPolicy.RUNTIME;
28  
29  import java.lang.annotation.Documented;
30  import java.lang.annotation.Retention;
31  import java.lang.annotation.Target;
32  
33  import javax.validation.Constraint;
34  import javax.validation.OverridesAttribute;
35  import javax.validation.Payload;
36  import javax.validation.ReportAsSingleViolation;
37  import javax.validation.constraints.Pattern;
38  
39  /**
40   * This class is basically a copy of {@link org.hibernate.validator.constraints.Email}.
41   * 
42   * The annotated element must be a valid email address. This class adds the field
43   * {@link #requiressuffix()}, thus only email adresses with a domain suffix will be valid. To
44   * achieve the behaviour of {@link org.hibernate.validator.constraints.Email} set this parameter to
45   * false.
46   *
47   * @author Christian Sterzl
48   * @since 1.2.6
49   */
50  @Documented
51  @Constraint(validatedBy = EmailValidator.class)
52  @Target({ METHOD, FIELD, ANNOTATION_TYPE, CONSTRUCTOR, PARAMETER })
53  @Retention(RUNTIME)
54  @ReportAsSingleViolation
55  @Pattern(regexp = "")
56  public @interface Email {
57    String message() default "{org.hibernate.validator.constraints.Email.message}";
58  
59    Class<?>[] groups() default {};
60  
61    Class<? extends Payload>[] payload() default {};
62  
63    /**
64     * Returns an additional regular expression the annotated string must match. The default is any
65     * string ('.*')
66     */
67    @OverridesAttribute(constraint = Pattern.class, name = "regexp")
68    String regexp() default ".*";
69  
70    /**
71     * Returns used in combination with {@link #regexp()} in order to specify a regular expression
72     * option.
73     */
74    @OverridesAttribute(constraint = Pattern.class, name = "flags")
75    Pattern.Flag[] flags() default {};
76  
77    /**
78     * Indicates whether or not a valid email adress requires a domain suffix.
79     */
80    boolean requiressuffix() default true;
81  
82    /**
83     * Defines several {@code @Email} annotations on the same element.
84     */
85    @Target({ METHOD, FIELD, ANNOTATION_TYPE, CONSTRUCTOR, PARAMETER })
86    @Retention(RUNTIME)
87    @Documented
88    public @interface List {
89      //CHECKSTYLE:OFF: 
90      Email[] value();
91      //CHECKSTYLE:ON:
92    }
93  }
